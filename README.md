# js-algorithms-and-data-structures

# Big O of Objects.
insertion - O(1)
Removal - O(1)
Searching - O(n)
Access - O(1)

# Big O of Object Methods.
Object.key - O(n)
Object.values - O(n)
Object.entries - O(n)
hasOwnProperty - O(1)

# Big O of Arrays.
Insertion - depends..
Removal - depends..
Searching - O(n)
Access - O(1)

# Big O of Array Methods.
push - O(1)
pop - O(1)
shift - O(n)
unshift - O(n)
concat - O(n)
slice - O(n)
splice - O(n)
sort - O(n * log n)
forEach/map/filter/reduce/etc. - O(n)
